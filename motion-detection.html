Motion Detection

Show more This is a real-time motion detection script using OpenCV's background subtraction. What it does: Opens the default webcam (index 0) Creates a MOG2 (Mixture of Gaussians) background subtractor, with shadow detection disabled for a cleaner binary mask In a continuous loop: Reads a frame from the camera Applies the background subtractor to get a foreground mask (pixels that differ from the learned background) Thresholds the mask to keep only strong foreground signals (values above 250) Cleans up noise using erosion (shrinks small white blobs) followed by dilation (regrows the remaining blobs, filling gaps) Finds contours (connected regions) in the cleaned mask Filters out small contours (area < 2000) to ignore noise/tiny movements For any contour that passes the size threshold, draws a green bounding box around it and flags movement = True If movement was detected, overlays a red "MOVEMENT DETECTED" text on the frame Displays both the annotated video feed and the raw mask in separate windows Exits when the Esc key (keycode 27) is pressed, then releases the camera and closes windows 


import cv2

camera = cv2.VideoCapture(0)

# background subtractor
backSub = cv2.createBackgroundSubtractorMOG2(
    detectShadows=False
)

while True:

    ret, frame = camera.read()

    # foreground mask
    mask = backSub.apply(frame)

    # threshold
    _, mask = cv2.threshold(
        mask,
        250,
        255,
        cv2.THRESH_BINARY
    )

    # noise removal
    mask = cv2.erode(mask, None, iterations=1)
    mask = cv2.dilate(mask, None, iterations=2)

    # contours
    contours, _ = cv2.findContours(
        mask,
        cv2.RETR_EXTERNAL,
        cv2.CHAIN_APPROX_SIMPLE
    )

    movement = False

    for contour in contours:

        # პატარა noise გამოტოვება
        if cv2.contourArea(contour) < 2000:
            continue

        movement = True

        x, y, w, h = cv2.boundingRect(contour)

        cv2.rectangle(
            frame,
            (x, y),
            (x + w, y + h),
            (0, 255, 0),
            2
        )

    if movement:

        cv2.putText(
            frame,
            "MOVEMENT DETECTED",
            (20, 40),
            cv2.FONT_HERSHEY_SIMPLEX,
            1,
            (0, 0, 255),
            2
        )

    cv2.imshow("Frame", frame)
    cv2.imshow("Mask", mask)

    key = cv2.waitKey(30)

    if key == 27:
        break

camera.release()
cv2.destroyAllWindows()
