Image Threshold Segmentation

This pipeline converts an image into grayscale, reduces noise, applies thresholding, and separates the main object from the background. 


import cv2


image = cv2.imread("image.jpg")

if image is None:
    print("Image not found.")
    exit()


gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)


blurred = cv2.GaussianBlur(
    gray,
    (5, 5),
    0
)


_, threshold_image = cv2.threshold(
    blurred,
    120,
    255,
    cv2.THRESH_BINARY
)


segmented_object = cv2.bitwise_and(
    image,
    image,
    mask=threshold_image
)


cv2.imwrite("threshold_result.jpg", threshold_image)
cv2.imwrite("segmented_object.jpg", segmented_object)

print("Results saved")


cv2.imshow("Original Image", image)
cv2.imshow("Threshold Segmentation", threshold_image)
cv2.imshow("Segmented Object", segmented_object)

cv2.waitKey(0)
cv2.destroyAllWindows()
