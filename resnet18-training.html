ResNet18 Image Classification Training

Used for: custom image classification via transfer learning, same goal as the VGG16 script — train a model to recognize your own categories from a folder of labeled images, instead of from-scratch training. Key difference from the VGG16 version: here the entire ResNet18 model is fine-tuned (all parameters stay trainable), not just the final layer. There's no freezing step like model.features.parameters().requires_grad = False, and the optimizer is given model.parameters() (the whole network) rather than just model.classifier.parameters(). Only the final fully-connected layer (model.fc) is swapped out to match the number of custom classes, but every layer is then updated during training. This is a more aggressive fine-tuning strategy: it can yield better accuracy than feature-extraction-only when you have enough data, since the pretrained features get adapted to your specific images, but it costs more compute and risks overfitting on small datasets (versus the frozen-backbone approach in the VGG16 script, which is faster and safer for small datasets). 


import torch
import torch.nn as nn
import torch.optim as optim

from torchvision import datasets, transforms, models
from torch.utils.data import DataLoader, random_split


# -------------------------------------------------
# ძირითადი პარამეტრები
# -------------------------------------------------

DATASET_PATH = "dataset"
MODEL_PATH = "resnet18_classifier.pth"

BATCH_SIZE = 4
EPOCHS = 5
LEARNING_RATE = 0.001
IMAGE_SIZE = 224


# -------------------------------------------------
# CPU ან GPU არჩევა
# -------------------------------------------------

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

print("Device:", device)


# -------------------------------------------------
# სურათების მომზადება
# -------------------------------------------------

transform = transforms.Compose([
    transforms.Resize((IMAGE_SIZE, IMAGE_SIZE)),
    transforms.ToTensor(),

    # ImageNet-ის ნორმალიზაცია საჭიროა pretrained ResNet18-ისთვის
    transforms.Normalize(
        mean=[0.485, 0.456, 0.406],
        std=[0.229, 0.224, 0.225]
    )
])


# -------------------------------------------------
# Dataset-ის ჩატვირთვა
# -------------------------------------------------

dataset = datasets.ImageFolder(
    root=DATASET_PATH,
    transform=transform
)

print("Classes:", dataset.classes)
print("Total images:", len(dataset))


# -------------------------------------------------
# Dataset-ის გაყოფა train და test ნაწილებად
# -------------------------------------------------

train_size = int(0.8 * len(dataset))
test_size = len(dataset) - train_size

train_dataset, test_dataset = random_split(
    dataset,
    [train_size, test_size]
)


# -------------------------------------------------
# DataLoader-ის შექმნა
# -------------------------------------------------

train_loader = DataLoader(
    train_dataset,
    batch_size=BATCH_SIZE,
    shuffle=True
)

test_loader = DataLoader(
    test_dataset,
    batch_size=BATCH_SIZE,
    shuffle=False
)


# -------------------------------------------------
# Pretrained ResNet18 მოდელის ჩატვირთვა
# -------------------------------------------------

num_classes = len(dataset.classes)

model = models.resnet18(
    weights=models.ResNet18_Weights.DEFAULT
)


# -------------------------------------------------
# ბოლო Fully Connected Layer-ის შეცვლა
# -------------------------------------------------

# ResNet18 თავდაპირველად 1000 ImageNet კლასზეა გაწვრთნილი.
# ამიტომ ბოლო layer უნდა შეიცვალოს ჩვენი dataset-ის კლასების რაოდენობით.

model.fc = nn.Linear(
    model.fc.in_features,
    num_classes
)

model = model.to(device)


# -------------------------------------------------
# Loss function და Optimizer
# -------------------------------------------------

criterion = nn.CrossEntropyLoss()

optimizer = optim.Adam(
    model.parameters(),
    lr=LEARNING_RATE
)


# -------------------------------------------------
# მოდელის გაწვრთნა
# -------------------------------------------------

for epoch in range(EPOCHS):
    model.train()

    total_loss = 0
    correct = 0
    total = 0

    for images, labels in train_loader:
        images = images.to(device)
        labels = labels.to(device)

        optimizer.zero_grad()

        outputs = model(images)

        loss = criterion(outputs, labels)

        loss.backward()

        optimizer.step()

        total_loss += loss.item()

        _, predicted = torch.max(outputs, 1)

        total += labels.size(0)
        correct += (predicted == labels).sum().item()

    train_accuracy = 100 * correct / total

    print(
        f"Epoch [{epoch + 1}/{EPOCHS}] "
        f"Loss: {total_loss:.4f} "
        f"Train Accuracy: {train_accuracy:.2f}%"
    )


# -------------------------------------------------
# Accuracy-ის გამოთვლა test dataset-ზე
# -------------------------------------------------

model.eval()

correct = 0
total = 0

with torch.no_grad():
    for images, labels in test_loader:
        images = images.to(device)
        labels = labels.to(device)

        outputs = model(images)

        _, predicted = torch.max(outputs, 1)

        total += labels.size(0)
        correct += (predicted == labels).sum().item()

test_accuracy = 100 * correct / total

print("Test Accuracy:", round(test_accuracy, 2), "%")


# -------------------------------------------------
# მოდელის შენახვა .pth ფორმატში
# -------------------------------------------------

torch.save(
    {
        "model_state_dict": model.state_dict(),
        "classes": dataset.classes,
        "test_accuracy": test_accuracy
    },
    MODEL_PATH
)

print("Model saved:", MODEL_PATH)
