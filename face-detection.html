Face Detection

Detects faces in real-time using Haar Cascade.


import cv2


# OpenCV-ს უკვე მოყვება Haar Cascade ფაილები.
# ეს classifier გამოიყენება სახეების ამოსაცნობად.
face_cascade = cv2.CascadeClassifier(
    cv2.data.haarcascades + "haarcascade_frontalface_default.xml"
)


# webcam-ის ჩართვა.
# 0 ნიშნავს ძირითად კამერას.
camera = cv2.VideoCapture(0)


# ვამოწმებთ, ჩაირთო თუ არა კამერა სწორად.
if not camera.isOpened():
    print("კამერა ვერ ჩაირთო.")
    exit()


while True:
    # კამერიდან ერთი frame-ის წაკითხვა
    ret, frame = camera.read()

    if not ret:
        print("კადრის წაკითხვა ვერ მოხერხდა.")
        break

    # Haar Cascade უკეთ მუშაობს grayscale სურათზე,
    # ამიტომ ფერად frame-ს გადავიყვანთ შავ-თეთრ ფორმატში.
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    # სახეების ამოცნობა
    faces = face_cascade.detectMultiScale(
        gray,
        scaleFactor=1.1,
        minNeighbors=5,
        minSize=(40, 40)
    )

    # ყველა ნაპოვნ სახეზე Bounding Box-ის დახატვა
    for (x, y, w, h) in faces:
        cv2.rectangle(
            frame,
            (x, y),
            (x + w, y + h),
            (0, 255, 0),
            2
        )

        cv2.putText(
            frame,
            "Face",
            (x, y - 10),
            cv2.FONT_HERSHEY_SIMPLEX,
            0.8,
            (0, 255, 0),
            2
        )

    # შედეგის რეალურ დროში ჩვენება
    cv2.imshow("Real Time Face Detection", frame)

    # q ღილაკზე დაჭერისას პროგრამა დასრულდება
    if cv2.waitKey(1) & 0xFF == ord("q"):
        break


# რესურსების გათავისუფლება
camera.release()
cv2.destroyAllWindows()
