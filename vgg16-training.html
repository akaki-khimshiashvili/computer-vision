VGG16 Transfer Learning

This VGG16 script is used for custom image classification — training a model to sort images into your own categories (not the standard ImageNet 1000 classes) by reusing a pretrained network's learned visual features rather than training from scratch. Practical use case: you'd point DATASET_PATH at a folder of your own labeled images (one subfolder per class — e.g., "cats", "dogs", "defective_parts", "plant_diseases", whatever the task is), and the script teaches VGG16 to recognize those specific categories. Since the convolutional layers stay frozen and only the final classifier is retrained, this works well even with a small dataset (the 80/20 split and small batch size of 4 suggest that's the expectation here), because you're leveraging features VGG16 already learned from millions of ImageNet images rather than learning visual patterns from zero. 


import torch
import torch.nn as nn
import torch.optim as optim

from torchvision import datasets, transforms, models
from torch.utils.data import DataLoader, random_split


# -------------------------------------------------
# ძირითადი პარამეტრები
# -------------------------------------------------

DATASET_PATH = "dataset"
MODEL_PATH = "vgg16_classifier.pth"

BATCH_SIZE = 4
EPOCHS = 5
LEARNING_RATE = 0.0001
IMAGE_SIZE = 224


# -------------------------------------------------
# CPU ან GPU არჩევა
# -------------------------------------------------

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

print("Device:", device)


# -------------------------------------------------
# სურათების მომზადება
# -------------------------------------------------

transform = transforms.Compose([
    transforms.Resize((IMAGE_SIZE, IMAGE_SIZE)),
    transforms.ToTensor(),

    # pretrained VGG16 გაწვრთნილია ImageNet-ზე.
    # ამიტომ ვიყენებთ ImageNet-ის normalization მნიშვნელობებს.
    transforms.Normalize(
        mean=[0.485, 0.456, 0.406],
        std=[0.229, 0.224, 0.225]
    )
])


# -------------------------------------------------
# Dataset-ის ჩატვირთვა
# -------------------------------------------------

dataset = datasets.ImageFolder(
    root=DATASET_PATH,
    transform=transform
)

print("Classes:", dataset.classes)
print("Total images:", len(dataset))


# -------------------------------------------------
# Dataset-ის გაყოფა Train და Test ნაწილებად
# -------------------------------------------------

train_size = int(0.8 * len(dataset))
test_size = len(dataset) - train_size

train_dataset, test_dataset = random_split(
    dataset,
    [train_size, test_size]
)


# -------------------------------------------------
# DataLoader-ის შექმნა
# -------------------------------------------------

train_loader = DataLoader(
    train_dataset,
    batch_size=BATCH_SIZE,
    shuffle=True
)

test_loader = DataLoader(
    test_dataset,
    batch_size=BATCH_SIZE,
    shuffle=False
)


# -------------------------------------------------
# Pretrained VGG16 მოდელის ჩატვირთვა
# -------------------------------------------------

num_classes = len(dataset.classes)

model = models.vgg16(
    weights=models.VGG16_Weights.DEFAULT
)


# -------------------------------------------------
# Feature extractor-ის გაყინვა
# -------------------------------------------------

# VGG16-ის convolution ნაწილი უკვე კარგად არის გაწვრთნილი ImageNet-ზე.
# პატარა dataset-ის შემთხვევაში უკეთესია თავიდან მხოლოდ classifier ნაწილი გავწვრთნათ.
for parameter in model.features.parameters():
    parameter.requires_grad = False


# -------------------------------------------------
# Classifier layer-ის შეცვლა
# -------------------------------------------------

# VGG16-ის ბოლო layer თავდაპირველად 1000 კლასზე მუშაობს.
# ჩვენ მას ვცვლით ჩვენი dataset-ის კლასების რაოდენობით.

model.classifier[6] = nn.Linear(
    in_features=model.classifier[6].in_features,
    out_features=num_classes
)

model = model.to(device)


# -------------------------------------------------
# Loss function და Optimizer
# -------------------------------------------------

criterion = nn.CrossEntropyLoss()

optimizer = optim.Adam(
    model.classifier.parameters(),
    lr=LEARNING_RATE
)


# -------------------------------------------------
# მოდელის გაწვრთნა
# -------------------------------------------------

for epoch in range(EPOCHS):
    model.train()

    total_loss = 0.0
    correct = 0
    total = 0

    for images, labels in train_loader:
        images = images.to(device)
        labels = labels.to(device)

        optimizer.zero_grad()

        outputs = model(images)

        loss = criterion(outputs, labels)

        loss.backward()

        optimizer.step()

        total_loss += loss.item()

        _, predicted = torch.max(outputs, 1)

        total += labels.size(0)
        correct += (predicted == labels).sum().item()

    train_accuracy = 100 * correct / total

    print(
        f"Epoch [{epoch + 1}/{EPOCHS}] "
        f"Loss: {total_loss:.4f} "
        f"Train Accuracy: {train_accuracy:.2f}%"
    )


# -------------------------------------------------
# Accuracy-ის გამოთვლა test dataset-ზე
# -------------------------------------------------

model.eval()

correct = 0
total = 0

with torch.no_grad():
    for images, labels in test_loader:
        images = images.to(device)
        labels = labels.to(device)

        outputs = model(images)

        _, predicted = torch.max(outputs, 1)

        total += labels.size(0)
        correct += (predicted == labels).sum().item()

test_accuracy = 100 * correct / total

print("Test Accuracy:", round(test_accuracy, 2), "%")


# -------------------------------------------------
# მოდელის შენახვა .pth ფორმატში
# -------------------------------------------------

torch.save(
    {
        "model_state_dict": model.state_dict(),
        "classes": dataset.classes,
        "test_accuracy": test_accuracy
    },
    MODEL_PATH
)

print("Model saved:", MODEL_PATH)
