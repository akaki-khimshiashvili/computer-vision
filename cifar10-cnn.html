CIFAR-10 CNN from Scratch

Used for: building and training an image classifier completely from scratch (no pretrained weights) on the CIFAR-10 benchmark dataset — 60,000 small 32x32 color images across 10 fixed classes (plane, car, bird, cat, etc., automatically downloaded via torchvision). How it differs from the VGG16/ResNet18 scripts: those used transfer learning on a custom dataset of your own images and a fixed number of arbitrary classes; this one defines its own small 3-block CNN architecture from zero and trains entirely on the standard CIFAR-10 dataset, with all weights randomly initialized rather than reused from ImageNet. It's a good baseline/from-scratch comparison point against the transfer-learning approaches in the other two scripts — useful for seeing how a small custom architecture performs without any pretrained head start. 


import torch
import torch.nn as nn
import torch.optim as optim

from torchvision import datasets, transforms
from torch.utils.data import DataLoader


# -------------------------------------------------
# ძირითადი პარამეტრები
# -------------------------------------------------

BATCH_SIZE = 64
EPOCHS = 10
LEARNING_RATE = 0.001
MODEL_PATH = "cifar10_cnn_model.pth"


# -------------------------------------------------
# CPU ან GPU არჩევა
# -------------------------------------------------

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

print("Device:", device)


# -------------------------------------------------
# CIFAR-10 სურათების მომზადება
# -------------------------------------------------

transform = transforms.Compose([
    transforms.ToTensor(),

    # CIFAR-10-ისთვის ხშირად გამოიყენება ეს normalization მნიშვნელობები.
    # ეს ეხმარება მოდელს უფრო სტაბილურად ისწავლოს.
    transforms.Normalize(
        mean=(0.4914, 0.4822, 0.4465),
        std=(0.2470, 0.2435, 0.2616)
    )
])


# -------------------------------------------------
# CIFAR-10 dataset-ის ჩამოტვირთვა
# -------------------------------------------------

train_dataset = datasets.CIFAR10(
    root="data",
    train=True,
    download=True,
    transform=transform
)

test_dataset = datasets.CIFAR10(
    root="data",
    train=False,
    download=True,
    transform=transform
)

print("Train images:", len(train_dataset))
print("Test images:", len(test_dataset))
print("Classes:", train_dataset.classes)


# -------------------------------------------------
# DataLoader-ის შექმნა
# -------------------------------------------------

train_loader = DataLoader(
    train_dataset,
    batch_size=BATCH_SIZE,
    shuffle=True
)

test_loader = DataLoader(
    test_dataset,
    batch_size=BATCH_SIZE,
    shuffle=False
)


# -------------------------------------------------
# CNN მოდელის შექმნა
# -------------------------------------------------

class CIFAR10CNN(nn.Module):
    def __init__(self):
        super().__init__()

        # CIFAR-10 სურათები არის 3x32x32 ზომის.
        # 3 ნიშნავს RGB არხებს.
        self.features = nn.Sequential(
            nn.Conv2d(
                in_channels=3,
                out_channels=32,
                kernel_size=3,
                padding=1
            ),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),

            nn.Conv2d(
                in_channels=32,
                out_channels=64,
                kernel_size=3,
                padding=1
            ),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),

            nn.Conv2d(
                in_channels=64,
                out_channels=128,
                kernel_size=3,
                padding=1
            ),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2)
        )

        # 32x32 სურათი სამი pooling layer-ის შემდეგ ხდება 4x4.
        # ბოლო feature map არის 128x4x4.
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128 * 4 * 4, 256),
            nn.ReLU(),
            nn.Linear(256, 10)
        )

    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)

        return x


model = CIFAR10CNN().to(device)


# -------------------------------------------------
# Loss function და Optimizer
# -------------------------------------------------

criterion = nn.CrossEntropyLoss()

optimizer = optim.Adam(
    model.parameters(),
    lr=LEARNING_RATE
)


# -------------------------------------------------
# მოდელის გაწვრთნა
# -------------------------------------------------

for epoch in range(EPOCHS):
    model.train()

    total_loss = 0.0
    correct = 0
    total = 0

    for images, labels in train_loader:
        images = images.to(device)
        labels = labels.to(device)

        optimizer.zero_grad()

        outputs = model(images)

        loss = criterion(outputs, labels)

        loss.backward()

        optimizer.step()

        total_loss += loss.item()

        _, predicted = torch.max(outputs, 1)

        total += labels.size(0)
        correct += (predicted == labels).sum().item()

    train_accuracy = 100 * correct / total

    print(
        f"Epoch [{epoch + 1}/{EPOCHS}] "
        f"Loss: {total_loss:.4f} "
        f"Train Accuracy: {train_accuracy:.2f}%"
    )


# -------------------------------------------------
# Accuracy-ის გამოთვლა test dataset-ზე
# -------------------------------------------------

model.eval()

correct = 0
total = 0

with torch.no_grad():
    for images, labels in test_loader:
        images = images.to(device)
        labels = labels.to(device)

        outputs = model(images)

        _, predicted = torch.max(outputs, 1)

        total += labels.size(0)
        correct += (predicted == labels).sum().item()

test_accuracy = 100 * correct / total

print("Test Accuracy:", round(test_accuracy, 2), "%")


# -------------------------------------------------
# მოდელის შენახვა .pth ფორმატში
# -------------------------------------------------

torch.save(
    {
        "model_state_dict": model.state_dict(),
        "classes": train_dataset.classes,
        "test_accuracy": test_accuracy
    },
    MODEL_PATH
)

print("Model saved as:", MODEL_PATH)
