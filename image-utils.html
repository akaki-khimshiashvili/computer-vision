Image Processing Utilities

Used for: simple object/background separation in a static image using intensity-based thresholding — a classical (non-ML) image segmentation technique, useful for cases where the object of interest is noticeably brighter or darker than the background (e.g., a clearly lit object against a dark backdrop). How it works: Loads image.jpg from disk, with a basic error check if the file isn't found Converts to grayscale (thresholding works on single-channel intensity values) Applies a light Gaussian blur to suppress small-scale noise before thresholding, so isolated stray pixels don't get misclassified as object/background Applies binary thresholding: any pixel above intensity 120 becomes white (255, treated as "object"), anything below becomes black (0, "background") Uses that binary mask with bitwise_and to extract just the object's pixels from the original color image, blacking out everything else Saves two outputs: the raw black-and-white threshold mask, and the color image with only the segmented object visible Displays all three images (original, threshold mask, segmented result) in windows until a key is pressed

What this includes

  • Image loading with validation
  • Gaussian noise generation
  • Median blur denoising
  • Gaussian blur smoothing
  • Full processing pipeline
  • Save & display helpers

import cv2


# -------------------------------------------------
# გამოსახულების წაკითხვა
# -------------------------------------------------

image = cv2.imread("image.jpg")

if image is None:
    print("გამოსახულება ვერ მოიძებნა.")
    print("დარწმუნდით, რომ image.jpg ფაილი იგივე საქაღალდეშია.")
    exit()


# -------------------------------------------------
# Grayscale ფორმატში გადაყვანა
# -------------------------------------------------

gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)


# -------------------------------------------------
# Noise-ის შემცირება
# -------------------------------------------------

# მცირე blur ეხმარება threshold-ს, რომ ზედმეტი პატარა წერტილები ნაკლებად გამოჩნდეს.
blurred = cv2.GaussianBlur(
    gray,
    (5, 5),
    0
)


# -------------------------------------------------
# Threshold Segmentation
# -------------------------------------------------

# threshold-ის შემდეგ პიქსელები იყოფა ორ ჯგუფად:
# 0   -> ფონი
# 255 -> ობიექტი
_, threshold_image = cv2.threshold(
    blurred,
    120,
    255,
    cv2.THRESH_BINARY
)


# -------------------------------------------------
# ობიექტის გამოყოფა ფონიდან
# -------------------------------------------------

# mask გამოიყენება იმისთვის, რომ ორიგინალ სურათში დარჩეს მხოლოდ ის ნაწილი,
# რომელიც threshold-ის მიხედვით ობიექტად ჩაითვალა.
segmented_object = cv2.bitwise_and(
    image,
    image,
    mask=threshold_image
)


# -------------------------------------------------
# შედეგის შენახვა
# -------------------------------------------------

cv2.imwrite("threshold_result.jpg", threshold_image)
cv2.imwrite("segmented_object.jpg", segmented_object)

print("შედეგები შენახულია:")
print("threshold_result.jpg")
print("segmented_object.jpg")


# -------------------------------------------------
# გამოსახულებების ჩვენება
# -------------------------------------------------

cv2.imshow("Original Image", image)
cv2.imshow("Threshold Segmentation", threshold_image)
cv2.imshow("Object Separated From Background", segmented_object)

cv2.waitKey(0)
cv2.destroyAllWindows()
